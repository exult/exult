%%section snap_zones
# Defines guided placement zones for ritual containers.
# When an item is dropped in a zone, it automatically snaps to the target position.
#
# Format: :gump_shape/zone_id/zone_x/zone_y/zone_w/zone_h/snap_x/snap_y/priority/zone_type
#
# gump_shape = Shape number in gumps.vga
# zone_id    = Unique identifier for this zone (descriptive, e.g., "north_candle")
# zone_x/y   = Top-left corner of detection area (relative to container_area)
# zone_w/h   = Width/height of detection area (defines the snap zone size)
# snap_x/y   = Target position where items snap to (relative to container_area)
# priority   = Selection priority for overlapping zones (higher = checked first)
# zone_type  = Category label (e.g., "candle", "reagent", "focus") - informational only
#
# Example - Pentagram ritual container (uncomment and modify as needed):
# :111/north_candle/20/0/20/20/30/10/3/candle
# :111/south_candle/55/110/20/20/65/120/3/candle
# :111/center_focus/55/55/20/20/65/65/4/focus
#
%%endsection

%%section shortcutbar_icon
# Set the icons for the shortcut bar buttons.
# Format: :slot/variant/select vga/shape/frame/extra frame/bool is_party_item/select fallback vga/fallback shape/fallback frame/fallback extra frame
# slots are 0/1/.../9. 0 is the first slot
# variant: 0 normal icon, 1 transparent icon, 2 found icon (displayed when is_party_item check succeeds)
#          (if a slot exists but no transparent icon is set, this icon will not be transparent
#          - if a patch gump_info.txt is used and the transparent icon is not set the default will be used)
# select .vga: 0 shortcutbar.vga, 1 shapes.vga, 2 gumps.vga, 3 paperdol.vga, -1 disables this slot, makes it invisible without a gap)
# frame/extra frame: frame of the shape to use. Extra frame is used if the action uses that (combat)
# bool is_party_item: if the shape is from shapes.vga it will check if the item is in the party - if not the slot is hidden or the fallback shape will be used if given.
# select fallback vga: same as select .vga.
# fallback shape: same as shape.
# fallback frame: same as shape.
# fallback extra frame: same as extra frame.
